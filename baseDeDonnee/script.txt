//Créer la base de donnée 
create database e2295517
//Créer la table utilisateurs (users)
create table users (
    userId int not null AUTO_INCREMENT,
    name varchar(25) not null,
    username varchar(45) not null,
    password varchar(20) not null,
    birthday date not null,
    CONSTRAINT pk_userId PRIMARY KEY(userId),
    CONSTRAINT unique_username UNIQUE(username)
    )

//Créer la table forum (articles)

create table articles (
    articleId int not null AUTO_INCREMENT,
    title varchar(100) not null,
    content text not null,
    date date not null,
    users_userId int not null,
    CONSTRAINT pk_articleId PRIMARY KEY(articleId),
    CONSTRAINT fk_users_userId FOREIGN KEY(users_userId) REFERENCES users(userId)
   )
